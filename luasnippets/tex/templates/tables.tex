% Tables
% https://tex.stackexchange.com/questions/12672/which-tabular-packages-do-which-tasks-and-which-packages-conflict
\usepackage{array} % flexible column formatting
\usepackage{tabularx} % Column type X for width filling.
\usepackage{tabulary} % Column type L,C,R,J for balanced width versions of l,c,r,j.
\usepackage{booktabs} % Better vertical spacing. Midrule etc with varying thickness instead of \hline.
\usepackage{colortbl} % coloured elements of table, e.g. gray lines (rules) to dim on any display.
\newcommand{\thinrule}{%
	\arrayrulecolor{gray}%
	% \specialrule{width}{above spacing}{below spacing}
	\specialrule{0.05ex}{0.1ex}{0.1ex}%
	\arrayrulecolor{black}
}
% creates missing tabularx column type named "R", and "C" to specify right and center adjustment of width-filling column.
% > is from array package. It inserts the {...} contents directly before each column.
% https://mirror.aarnet.edu.au/pub/CTAN/macros/latex/required/tools/array.pdf
\newcolumntype{R}{>{\raggedleft\arraybackslash}X}
\newcolumntype{C}{>{\centering\arraybackslash}X}
\usepackage{multicol, multirow} % Also see pbox.
