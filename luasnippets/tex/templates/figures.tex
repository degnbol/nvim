% https://tex.stackexchange.com/questions/8351/what-do-makeatletter-and-makeatother-do#8353
\makeatletter
% set h(ere), t(op) as default figure placement.
% ! makes it try harder to place here.
\renewcommand*{\fps@figure}{ht!}
\makeatother

% set textwidth as default includegraphics width.
\let\RAWincludegraphics\includegraphics
\renewcommand{\includegraphics}[2][]{\RAWincludegraphics[width=\textwidth,#1]{#2}}

% import pdf with a pdf_tex file with command \pdftex[\textwidth]{folder/}{filename.pdf}
% imported files created from svg that you can make with affinity designer, 
% then convert with shell alias `pdftex filename.svg`
\newcommand{\pdftex}[3][\columnwidth]{
	\def\svgwidth{#1}
	\sans{% sans-serif fonts in figures
		\import{#2}{#3_tex}
	}
}
% use the same command to set textsize for subcaption text in figures.
\newcommand{\subsize}[1]{\Large{#1}}
